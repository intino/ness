keytool -genkey -alias server -keyalg RSA -keysize 2048 -dname "CN=localhost,OU=IT,O=Company,L=LPA,S=LPA,C=ES" -keystore server.jks -storepass {server_password};

keytool -genkey -alias client -keyalg RSA -keysize 2048 -dname "CN=localhost,OU=IT,O=Company,L=LPA,S=LPA,C=ES" -keystore client.jks -storepass {client_password};


keytool -export -alias server -keystore server.jks -file server.crt -storepass {server_password};
keytool -export -alias client -keystore client.jks -file client.crt -storepass {client_password};

keytool -import -alias server -keystore client.jts -file server.crt -storepass {server_password};
keytool -import -alias client -keystore server.jts -file client.crt -storepass {client_password};
